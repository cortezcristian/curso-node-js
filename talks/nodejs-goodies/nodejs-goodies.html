NODE.JS

Telenet Chat Server

var net = require('net'), sockets = [];

var server = net.createServer(function (socket) {
      sockets.push(socket);
      socket.write('Bienvenido al Chat ;)\r\n');
      
      socket.on("data", function(d){
        sockets.forEach(function(v,i){
            v.write(d);    
        });    
      });

      socket.on("end", function(){
          var index = sockets.indexof(socket);
          delete sockets[index];
      });
}); server.listen(1337, '127.0.0.1');

Automatizando con Grunt


Problema: procesar archivos less

$ lessc -x estilos.less estilos.css # comando manual

Paso 1: Creamos un package.json

$ cat package.json
{
 "name": "pre-procesar-less",
 "version": "0.0.1",
 "devDependencies": {
   "grunt": "~0.4.2",
   "grunt-contrib-less": "*"
 }
}

Automatizando con Grunt


Paso 2: Creamos el Gruntfile.js

$ cat Gruntfile.js
module.exports = function(grunt) {
 
grunt.initConfig({
   pkg: grunt.file.readJSON('package.json'),
   less: {
     process: {
       options: {compress:true},
       files: {
         'estilos.css': 'estilos.less',
       }
     }
   }
 }); // end grunt.initConfig
 
 grunt.loadNpmTasks('grunt-contrib-less'); // Grunt Package
};

Automatizando con Grunt


¿Cómo usarlo?

$ npm install # primero instalar las dependencias

Corremos la tarea:

$ grunt less:process

Automatizando con Grunt


Hasta este punto solo hemos reemplazado esto:

$ lessc -x estilos.less estilos.css # comando manual

por esto:

$ grunt less:process # grunt task

Automatizando con Grunt


Problema: automatizar el procesamiento de archivos less, de manera que al modificar un archivo .less el archivo estilos.css se genere automáticamente

Paso 1: Agregar una nueva dependencia package.json

$ cat package.json
{
 "name": "pre-procesar-less",
 "version": "0.0.1",
 "devDependencies": {
   "grunt": "~0.4.2",
   "grunt-contrib-less": "*",
   "grunt-contrib-watch": "*"
 }
}

Automatizando con Grunt

Paso 2: Modificamos nuestro Grunfile.js

$ cat Gruntfile.js 
module.exports = function(grunt) {
grunt.initConfig({
   pkg: grunt.file.readJSON('package.json'),
   less: {
     /* [ ... ] idem ejemplo aterior */
   },
   watch: {
     lessfiles: {
       files: '*.less',
       tasks: ['less:process']
     }
   }
 }); // end grunt.initConfig
 grunt.loadNpmTasks('grunt-contrib-less');
 grunt.loadNpmTasks('grunt-contrib-watch'); // <- nueva dependencia
};

Automatizando con Grunt


¿Cómo usarlo?

$ npm install # primero instalar las nuevas dependencias

Abrimos una nueva terminal y corremos la tarea:

Express

Supongamos que es necesario tener una cookie activa para poder acceder al sitio:


var cookieAuthorizer = function(req, res, next) {
    if(typeof req.cookies.authorizeme == "undefined"){
        res.redirect('/site-not-available');         
    } else {
        next();
    }
}

app.get('/', cookieAuthorizer, function(req, res){
    /**
    * Esta vista solo se muestra si el usuario tiene la cookie
    */
    res.render('index', { title: 'Express'}); 
});
